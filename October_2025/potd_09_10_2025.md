## Postorder Traversal
### Java Solution
```java
/*
class Node {
    int data;
    Node left, right;
    Node(int val){
        data = val;
        left = right = null;
    }
}
*/

import java.util.*;

class Solution {
    // Function to return the postorder traversal of a binary tree
    public ArrayList<Integer> postOrder(Node root) {
        ArrayList<Integer> result = new ArrayList<>();
        postOrderHelper(root, result);
        return result;
    }

    // Helper function to perform recursive postorder traversal
    private void postOrderHelper(Node root, ArrayList<Integer> result) {
        // Base case: if the node is null, do nothing
        if (root == null)
            return;

        // Step 1: Recurse on the left subtree
        postOrderHelper(root.left, result);

        // Step 2: Recurse on the right subtree
        postOrderHelper(root.right, result);

        // Step 3: Visit the node (add data to result list)
        result.add(root.data);
    }
}
```

### C++ Solution
```cpp
/*
class Node {
  public:
    int data;

    Node* right;

        left = NULL;

    }
};

class Solution {
  public:
    // Function to return the postorder traversal of a binary tree
    vector<int> postOrder(Node* root) {
        vector<int> result;
        postOrderHelper(root, result);
        return result;
    }

  private:
    // Recursive helper function
    void postOrderHelper(Node* root, vector<int>& result) {
        // Base case: if node is null, do nothing
        if (root == nullptr)
            return;

        // Step 1: Recurse on the left subtree
        postOrderHelper(root->left, result);

        // Step 2: Recurse on the right subtree
        postOrderHelper(root->right, result);

        // Step 3: Process the current node
        result.push_back(root->data);
    }
};
```

### Python Solution
```py
'''
class Node:
    def __init__(self, val):
        self.data = val
        self.left = None
        self.right = None
'''

class Solution:
    def postOrder(self, root):
        # This list will store the result of postorder traversal
        result = []
        
        # Helper recursive function
        def dfs(node):
            # Base case: if node is None, return
            if not node:
                return
            
            # Step 1: Traverse left subtree
            dfs(node.left)
            
            # Step 2: Traverse right subtree
            dfs(node.right)
            
            # Step 3: Visit current node
            result.append(node.data)
        
        # Call the helper function starting from root
        dfs(root)
        
        # Return the final traversal list
        return result
```
